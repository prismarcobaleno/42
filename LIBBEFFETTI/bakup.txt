int	ft_isalpha(char c)
{
	if ((c < 'a' || c > 'z') && (c < 'A' || c > 'Z'))
	{
		return (0);
	}
	return (1);
}

int	ft_isdigit(int c)
{
	if ((c < '0' || c > '9'))
	{
		return (0);
	}
	return (1);
}

int	ft_isalnum(char c)
{
	if ((c >= '0' && c <= '9'))
	{
		return (1);
	}
	else if ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z'))
	{
		return (1);
	}
	return (0);
}


int	ft_isascii(char c)
{
	if (c >= 0 && c <= 127)
	{
		return (1);
	}
	return (0);
}

int	ft_isprint(char c)
{
	if (c < 32 || c > 126)
	{
		return (0);
	}
	return (1);
}

int	ft_strlen(char *str)
{
	int	i;

	i = 0;
	while (str[i])
	{
		i += 1;
	}
	return (i);
}

//serve per la prossima
int	se_minuscola(int s)
{
	if (!(s >= 'a' && s <= 'z'))
		return (0);
	else
		return (1);
}

int	ft_toupper(int c)
{
	if (se_minuscola(c) == 1)
		c = c - 32;
	return (c);
}

//serve per la prossima
int	se_maiuscola(int s)
{
	if (!(s >= 'A' && s <= 'Z'))
		return (0);
	else
		return (1);
}

int	ft_tolower(int c)
{
	if (se_maiuscola(c) == 1)
		c = c + 32;
	return(c);
}

void	*ft_memset(void *str, int c, unsigned long int n)
{
	char *p;
	unsigned int i;
	
	p = (char *)str;
	i = 0;
	while (i < n)
	{
		p[i] = c;
		i++;
	}
	return(str);
}


void	ft_bzero(void *s, unsigned long int n)
{
	char *p = (char *)s;
	unsigned int i = 0;
	
	while (i < n)
	{
		p[i] = '\0';
		i++;
	}
}

void	*ft_memcpy(void *dest, const void *src, unsigned long int n)
{
	unsigned int	i;
	char *fonte;
	char *destino;
	
	fonte = (char *)src;
	destino = (char *)dest;

	i = 0;
	while ((fonte[i] != '\0') && (i < n))
	{
		destino[i] = fonte[i];
		i++;
	}
		return (dest);
}

void	*ft_memmove(void *dest, const void *src, unsigned long int n)
{
	unsigned int	i;
	char *fonte;
	char *destino;
	char *tempo;

	i = 0;
	fonte = (char *)src;
	destino = (char *)dest;
	tempo = (char *)malloc(sizeof(char) * n);

	if(tempo == NULL)
	{
		return NULL;
	}
	else
	{
		while ((fonte[i] != '\0') && (i < n))
		{
			tempo[i] = fonte[i];
			i++;
		}
		i = 0;
		while ((tempo[i] != '\0') && (i < n))
		{
			destino[i] = tempo[i];
			i++;
		}
		free(tempo);
	}
	return (dest);
}

int	ft_strncmp(char *s1, const char *s2, unsigned long int n)
{
	unsigned long int	i;
	unsigned long int conta;

	i = 0;
	conta = 0;
	while (conta == 0 && n-- > 0)
	{
		if (s1[i] > s2[i])
		{
			conta = s1[i] - s2[i];
		}
		else if (s1[i] < s2[i])
		{
			conta = -(s2[i] - s1[i]);
		}
		if (s1[i] == '\0')
		{
			return (conta);
		}
		i++;
	}
	return (conta);
}

int	se_spazio(char s)
{
	if (s >= '\t' && s <= '\r')
		return (1);
	else if (s == 32)
		return (1);
	else
		return (0);
}

int	ft_atoi(const char *nptr)
{
	int	neg;
	int	num;
	int	i;

	i = 0;
	neg = 1;
	num = 0;
	while (se_spazio(nptr[i]))
		i++;
	if (nptr[i] == '-' || nptr[i] == '+')
	{
		if (nptr[i] == '-')
		{
			neg *= -1;
		}
		i++;
	}
	while (nptr[i] >= '0' && nptr[i] <= '9')
	{
		num = num * 10 + (nptr[i] - 48);
		i++;
	}
	return (num * neg);
}

void	*ft_memchr(const void *s, int c, unsigned long n)
{
	unsigned char	cara;
	unsigned char	*str;
	unsigned long	i;

	cara = c;
	i = 0;
	str = (unsigned char *) s;
	while (i++ < n && str[i] != '\0')
	{
		if (str[i] == cara)
		{
			return (&str[i]);
		}
	}
	return (NULL);
}
